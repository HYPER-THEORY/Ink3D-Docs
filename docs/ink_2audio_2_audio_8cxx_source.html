33Audio::Audio(const std::string& p) {
34 if (SDL_LoadWAV(p.c_str(), &spec, &buffer, &length) == nullptr) {
35 auto error = SDL_GetError();
36 printf("%s\n", error);
37 Error::set("Audio: Error reading from WAVE file");
38 }
39 ratio = spec.channels * spec.freq * (spec.format >> 3 & 0x1F);
40 spec.userdata = this;
41 spec.callback = [](void* u, uint8_t* s, int l) -> void {
42 auto* a = static_cast<Audio*>(u);
43 uint8_t* buffer = a->buffer + a->position;
44 uint32_t length = 0;
45 uint32_t volume = SDL_MIX_MAXVOLUME * a->volume;
46 if (a->position + l < a->length) {
47 length = l;
48 a->position += l;
49 } else {
50 length = a->length - a->position;
51 a->position = 0;
52 if (!a->loop) SDL_PauseAudioDevice(a->device, 1);
53 }
54 SDL_memset(s, 0, l);
55 SDL_MixAudioFormat(s, buffer, a->spec.format, length, volume);
56 };
57 device = SDL_OpenAudioDevice(nullptr, 0, &spec, nullptr, 0);
58}
