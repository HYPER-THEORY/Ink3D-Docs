31int Loader::load_image(const std::string& f, Image& i, bool s) {
32 SDL_Surface* surface = IMG_Load(f.c_str());
33 if (!surface) return set_error("[Loader Error] Error reading from image");
34 i.width = surface->w;
35 i.height = surface->h;
36 i.bytes = surface->format->BytesPerPixel;
37 i.data.resize(i.width * i.height * i.bytes);
38 SDL_memcpy(i.data.data(), surface->pixels, i.data.size());
39 SDL_FreeSurface(surface);
40 if (s) i.swap_rb();
41 return 0;
42}

44int Loader::load_obj(const std::string& f, Mesh* m) {
45 std::ifstream in;
46 in.open(f, std::ifstream::in);
47 if (!in) return set_error("[Loader Error] Error reading from obj file");
48 /* temporary data */
49 std::vector<Vec3> vertex;
50 std::vector<Vec3> normal;
51 std::vector<Vec2> uv;
52 /* initialize mesh object */
53 Mesh* current_mesh = m;
54 *current_mesh = Mesh();
55 current_mesh->name = "default";
56 /* initialize material */
57 MeshGroup* current_material = nullptr;
58 /* initialize group */
59 current_mesh->groups.push_back({"default", 0, 0});
60 MeshGroup* current_group = &current_mesh->groups.back();
61 /* initialize total length */
62 int total_length = 0;
63 /* read data by line */
64 std::string line;
65 while (!in.eof()) {
66 std::getline(in, line);
67 std::istringstream stream(line.c_str());
68 std::string keyword;
69 stream >> keyword;
70 if (keyword == "v") {
71 /* add vertex to temporary array */
72 Vec3 v;
73 stream >> v.x >> v.y >> v.z;
74 vertex.push_back(v);
75 } else if (keyword == "vn") {
76 /* add normal to temporary array */
77 Vec3 vn;
78 stream >> vn.x >> vn.y >> vn.z;
79 normal.push_back(vn);
80 } else if (keyword == "vt") {
81 /* add uv to temporary array */
82 Vec2 vt;
83 stream >> vt.x >> vt.y;
84 uv.push_back(vt);
85 } else if (keyword == "f") {
86 /*
87 * search for data by index and add it to current_mesh
88 * increase the length of current_material, current_group
89 */
90 int index;
91 for (int i = 0; i < 3; ++i) {
92 stream >> index;
93 current_mesh->vertex.push_back(vertex[index - 1]);
94 stream.get();
95 /* whether uv is omitted */
96 if (std::isdigit(stream.peek())) {
97 stream >> index;
98 current_mesh->uv.push_back(uv[index - 1]);
99 }
100 stream.get();
101 stream >> index;
102 current_mesh->normal.push_back(normal[index - 1]);
103 }
104 if (current_material) {
105 current_material->length += 3;
106 } else {
107 set_error("[Loader Error] Material is not assigned");
108 }
109 current_group->length += 3;
110 total_length += 3;
111 } else if (keyword == "g") {
112 /* add new group */
113 std::string name;
114 stream >> name;
115 /* if default group has no data, just replace its name */
116 if (current_group->name == "default" && current_group->length == 0) {
117 current_group->name = name;
118 } else {
119 /* new group */
120 current_mesh->groups.push_back({name, total_length, 0});
121 current_group = &current_mesh->groups.back();
122 }
123 } else if (keyword == "o") {
124 /* create new mesh object and initialize everything */
125 std::string name;
126 stream >> name;
127 /* if default mesh data has no data, just replace its name */
128 if (current_mesh->name == "default" && vertex.empty()) {
129 current_mesh->name = name;
130 } else {
131 /* new mesh object */
132 *(++current_mesh) = Mesh();
133 current_mesh->name = name;
134 /* initialize total length */
135 total_length = 0;
136 /* initialize group */
137 current_mesh->groups.push_back({"default", 0, 0});
138 current_group = &current_mesh->groups.back();
139 /* initialize material */
140 current_material = nullptr;
141 }
142 } else if (keyword == "usemtl") {
143 /* use material */
144 std::string name;
145 stream >> name;
146 /* new material reference */
147 current_mesh->material_groups.push_back({name, total_length, 0});
148 current_material = &current_mesh->material_groups.back();
149 /* update material library */
150 if (current_mesh->material_library.count(name) == 0) {
151 current_mesh->material_library.insert({name, nullptr});
152 }
153 }
154 }
155 in.close();
156 return static_cast<int>(current_mesh - m + 1);
157}

159int Loader::load_mtl(const std::string& f, SimpleMat* m) {
160 std::ifstream in;
161 in.open(f, std::ifstream::in);
162 if (!in) return set_error("[Loader Error] Error reading from mtl file");
163 std::string line;
164 SimpleMat* current_material = m - 1;
165 while (!in.eof()) {
166 std::getline(in, line);
167 std::istringstream stream(line.c_str());
168 std::string keyword;
169 stream >> keyword;
170 if (keyword == "newmtl") {
171 *(++current_material) = SimpleMat();
172 stream >> current_material->name;
173 } else if (current_material < m) {
174 /* material is not declared */
175 } else if (keyword == "Kd") {
176 Vec3& color = current_material->color;
177 stream >> color.x >> color.y >> color.z;
178 }
179 }
180 in.close();
181 return static_cast<int>(current_material - m + 1);
182}
