61int run() {
62 Settings t;
63 conf(t); /* config setting */
64
65 /* initialize window with parameters */
66 Ink::Legacy::Window::init(t.title, t.x, t.y, t.width, t.height, t.opengl, t.highdpi);
67 if (t.opengl) {
68 Ink::Legacy::Window::init_opengl(t.vsync, t.depth, t.stencil, t.msaa);
69 } else {
70 Ink::Legacy::Window::init_canvas();
71 frame.resize(t.width * t.height);
72 }
73 if (t.fullscreen) {
74 Ink::Legacy::Window::fullscreen_desktop();
75 }
76 if (t.hide_cursor) {
77 Ink::Legacy::Window::hide_cursor();
78 }
79 if (t.lock_cursor) {
80 Ink::Legacy::Window::lock_cursor();
81 Ink::Legacy::Window::move_cursor(t.width / 2, t.height / 2);
82 }
83 Ink::Legacy::Window::set_borderless(t.borderless);
84 Ink::Legacy::Window::set_resizable(t.resizable);
85 Ink::Legacy::Window::set_min_size(t.min_width, t.min_height);
86 Ink::Legacy::Window::set_interval(1000 / t.fps);
87
88 /* print error from Ink3D */
89 Ink::Error::set_callback([](const std::string& s) -> void {
90 std::cerr << Ink::Date::get_local().format() << ' ' << s << '\n';
91 });
92
93 load(); /* load once */
94
95 /* main loop */
96 while (Ink::Legacy::Window::is_open()) {
97
98 /* update frame with OpenGL */
99 if (t.opengl) {
100 Ink::Gpu::FrameBuffer::activate(nullptr);
101 Ink::Gpu::set_clear_color(t.background_color, 0);
102 Ink::Gpu::clear();
103 update(Ink::Legacy::Window::get_deltatime() * 0.001f);
104 }
105
106 /* update frame without OpenGL */
107 else {
108 std::fill(frame.begin(), frame.end(), Ink::Vec4(t.background_color, 0));
109 update(Ink::Legacy::Window::get_deltatime() * 0.001f);
110 uint32_t* canvas = Ink::Legacy::Window::get_canvas();
111 int pixel = t.width * t.height;
112 while (pixel --> 0) {
113 canvas[pixel] = roundf(frame[pixel].x * 0xFF) * 0x010000 +
114 roundf(frame[pixel].y * 0xFF) * 0x000100 +
115 roundf(frame[pixel].z * 0xFF) * 0x000001;
116 }
117 }
118
119 /* update window */
120 Ink::Legacy::Window::update();
121
122 /* print error from OpenGL */
123 if (t.opengl) {
124 std::string error = Ink::Gpu::get_error();
125 if (!error.empty()) std::cerr << error;
126 }
127
128#ifdef DEBUG
129 /* close window when pressing escape key */
130 if (Ink::Legacy::Window::is_down(SDLK_ESCAPE)) {
131 Ink::Legacy::Window::close();
132 }
133#endif
134 }
135
136 quit(); /* quit window */
137
138 return 0;
139}
