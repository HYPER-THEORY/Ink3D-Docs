54void SSAOPass::render() const {
55 /* fetch SSAO shader from shader lib */
56 Defines ssao_defines;
57 ssao_defines.set("SAMPLES", std::to_string(samples));
58 auto* ssao_shader = ShaderLib::fetch("SSAO", ssao_defines);
59
60 /* fetch blur shader from shader lib */
61 Defines blur_defines;
62 blur_defines.set("BLUR_BILATERAL");
63 blur_defines.set("TYPE", "float");
64 blur_defines.set("SWIZZLE", ".x");
65 auto* blur_shader = ShaderLib::fetch("Blur", blur_defines);
66
67 /* fetch blend shader from shader lib */
68 Defines blend_defines;
69 blend_defines.set("BLEND_OP(a, b)", "a * b");
70 blend_defines.set("A_SWIZZLE", ".xyzw");
71 blend_defines.set("B_SWIZZLE", ".xxxx");
72 auto* blend_shader = ShaderLib::fetch("Blend", blend_defines);
73
74 /* calculate camera & screen parameters */
75 Mat4 view_proj = camera->projection * camera->viewing;
76 Mat4 inv_view_proj = inverse_4x4(view_proj);
77 Vec2 screen_size = Vec2(width / 2, height / 2);
78
79 /* change the current viewport */
80 Gpu::Rect viewport = RenderPass::get_viewport();
81 RenderPass::set_viewport(Gpu::Rect(width / 2, height / 2));
82
83 /* 1. render SSAO to texture (down-sampling) */
84 ssao_shader->use_program();
85 ssao_shader->set_uniform_f("radius", radius);
86 ssao_shader->set_uniform_f("max_radius", max_radius);
87 ssao_shader->set_uniform_f("bias", bias);
88 ssao_shader->set_uniform_f("intensity", intensity);
89 ssao_shader->set_uniform_f("camera_near", camera->near);
90 ssao_shader->set_uniform_f("camera_far", camera->far);
91 ssao_shader->set_uniform_m4("view_proj", view_proj);
92 ssao_shader->set_uniform_m4("inv_view_proj", inv_view_proj);
93 ssao_shader->set_uniform_i("buffer_n", buffer_n->activate(0));
94 ssao_shader->set_uniform_i("buffer_d", buffer_d->activate(1));
95 RenderPass::render_to(ssao_shader, blur_target_1.get());
96
97 /* 2. blur texture for two times */
98 for (int i = 0; i < 2; ++i) {
99
100 /* blur texture horizontally */
101 blur_shader->use_program();
102 blur_shader->set_uniform_v2("direction", Vec2(1 / screen_size.x, 0));
103 blur_shader->set_uniform_i("radius", 7);
104 blur_shader->set_uniform_f("sigma_s", 2);
105 blur_shader->set_uniform_f("sigma_r", 0.25);
106 blur_shader->set_uniform_i("map", blur_map_1->activate(0));
107 RenderPass::render_to(blur_shader, blur_target_2.get());
108
109 /* blur texture vertically */
110 blur_shader->use_program();
111 blur_shader->set_uniform_v2("direction", Vec2(0, 1 / screen_size.y));
112 blur_shader->set_uniform_i("radius", 7);
113 blur_shader->set_uniform_f("sigma_s", 2);
114 blur_shader->set_uniform_f("sigma_r", 0.25);
115 blur_shader->set_uniform_i("map", blur_map_2->activate(0));
116 RenderPass::render_to(blur_shader, blur_target_1.get());
117 }
118
119 /* set back to the initial viewport */
120 RenderPass::set_viewport(viewport);
121
122 /* 3. render results to render target (up-sampling) */
123 blend_shader->use_program();
124 blend_shader->set_uniform_i("map_a", map->activate(1));
125 blend_shader->set_uniform_i("map_b", blur_map_1->activate(0));
126 RenderPass::render_to(blend_shader, target);
127}
