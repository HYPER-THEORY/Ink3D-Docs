207template <typename t>
208bool* List<t>::add(t o) {
209 objects.push_back({o, true});
210 return &objects.back().valid;
211}

213template <typename t>
214void List<t>::remove() {

216 o->valid ? ++o : o = objects.erase(o);
217 }
