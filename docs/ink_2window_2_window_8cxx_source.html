32void Window::init(const std::string& t, int x, int y, int w, int h, bool d) {
33 /* whether to center the window */
34 if (x == -1) x = SDL_WINDOWPOS_CENTERED;
35 if (y == -1) y = SDL_WINDOWPOS_CENTERED;

37 /* initialize keydown with false */

40 /* create a window in SDL system */

42 uint32_t flags = SDL_WINDOW_SHOWN | SDL_WINDOW_OPENGL;
43 if (d) flags |= SDL_WINDOW_ALLOW_HIGHDPI;
44 sdl_window = SDL_CreateWindow(t.c_str(), x, y, w, h, flags);

47void Window::init_opengl(int v, int d, int s, int m, bool a) {
48 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);

69void Window::update() {
70 /* init key pressed & released array */
71 std::fill_n(keypressed, 512, false);
72 std::fill_n(keyreleased, 512, false);
73
74 /* handle events */
75 SDL_Event event;
76 while (SDL_PollEvent(&event)) {
77 int32_t keycode = event.key.keysym.sym;
78 if (keycode > 127) keycode -= 1073741696;
79 if (event.type == SDL_QUIT) {
80 close();
81 return;
82 } else if (event.type == SDL_KEYDOWN) {
83 keypressed[keycode] = !keydown[keycode];
84 keydown[keycode] = true;
85 } else if (event.type == SDL_KEYUP) {
86 keyreleased[keycode] = keydown[keycode];
87 keydown[keycode] = false;
88 } else if (event.type == SDL_MOUSEBUTTONDOWN) {
89 if (event.button.button == SDL_BUTTON_LEFT) {
90 keypressed[1] = !keydown[1];
91 keydown[1] = true;
92 } else if (event.button.button == SDL_BUTTON_RIGHT) {
93 keypressed[2] = !keydown[2];
94 keydown[2] = true;
95 }
96 } else if (event.type == SDL_MOUSEBUTTONUP) {
97 if (event.button.button == SDL_BUTTON_LEFT) {
98 keyreleased[1] = keydown[1];
99 keydown[1] = false;
100 } else if (event.button.button == SDL_BUTTON_RIGHT) {
101 keyreleased[2] = keydown[2];
102 keydown[2] = false;
103 }
104 } else if (event.type == SDL_MOUSEMOTION) {
105 cursor_x = event.motion.x;
106 cursor_y = event.motion.y;
107 }
108 }
109
110 /* lock cursor to the center of window */
111 if (cursor_locked && SDL_GetKeyboardFocus()) {
112 std::pair<int, int> size;
113 SDL_GetWindowSize(sdl_window, &size.first, &size.second);
114 SDL_WarpMouseInWindow(sdl_window, size.first / 2, size.second / 2);
115 }
116
117 /* refresh window */
118 SDL_GL_SwapWindow(sdl_window);
119
120 /* calculate delta time */
121 uint32_t deltatime = SDL_GetTicks() - time;
122 if (deltatime <= interval) SDL_Delay(interval - deltatime);
123 time = SDL_GetTicks();
124}
