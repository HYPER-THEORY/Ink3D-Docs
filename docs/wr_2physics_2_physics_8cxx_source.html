46bool Cuboid::hittest(const Cuboid& c) {
47 return contain(c.v1.x, c.v1.y, c.v1.z) || contain(c.v1.x, c.v1.y, c.v2.z) ||
48 contain(c.v1.x, c.v2.y, c.v1.z) || contain(c.v1.x, c.v2.y, c.v2.z) ||
49 contain(c.v2.x, c.v1.y, c.v1.z) || contain(c.v2.x, c.v1.y, c.v2.z) ||
50 contain(c.v2.x, c.v2.y, c.v1.z) || contain(c.v2.x, c.v2.y, c.v2.z) ||
51 c.contain(v1.x, v1.y, v1.z) || c.contain(v1.x, v1.y, v2.z) ||

74void Solid::collide(Solid* s, float x, float y, float z) {
75 Cuboid& collided = s->box;
76 if (s != this && box.hittest(collided)) {
77 if (x != 0) position.x = x > 0 ? collided.v1.x - width - buffer : collided.v2.x + buffer;
78 if (y != 0) position.y = y > 0 ? collided.v1.y - height - buffer : collided.v2.y + buffer;
79 if (z != 0) position.z = z > 0 ? collided.v1.z - depth - buffer : collided.v2.z + buffer;
80 refresh();
81 }
82}

84void Solid::move(const Vec3& d) {
85 world.remove();
86 position.x += d.x;
87 refresh();
88 for (size_t i = world.size(); i --> 0;) {
89 collide(world[i], d.x, 0, 0);
90 }
91 position.y += d.y;
92 refresh();
93 for (size_t i = world.size(); i --> 0;) {
94 collide(world[i], 0, d.y, 0);
95 }
96 position.z += d.z;
97 refresh();
98 for (size_t i = world.size(); i --> 0;) {
99 collide(world[i], 0, 0, d.z);
100 }
101}
