27void Window::init() {
28 opening = true;
29 SDL_Init(SDL_INIT_VIDEO);
30 uint32_t flags = 0;
31 if (borderless) flags |= SDL_WINDOW_BORDERLESS;
32 if (resizable) flags |= SDL_WINDOW_RESIZABLE;
33 if (highdpi) flags |= SDL_WINDOW_ALLOW_HIGHDPI;
34 if (opengl) {
35 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
36 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 4);
37 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 1);
38 SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, 24);
39 SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, 8);
40 SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
41 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, msaa);
42 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, msaa_samples);
43 SDL_GL_SetAttribute(SDL_GL_ACCELERATED_VISUAL, accelerated);
44 flags |= SDL_WINDOW_OPENGL;
45 sdl_window = SDL_CreateWindow(title.c_str(), x, y, width, height, flags);
46 context = SDL_GL_CreateContext(sdl_window);
47 } else {
48 flags |= SDL_WINDOW_SHOWN;
49 sdl_window = SDL_CreateWindow(title.c_str(), x, y, width, height, flags);
50 surface = SDL_GetWindowSurface(sdl_window);
51 canvas_vector = std::vector<uint32_t>(width * height);
52 canvas = canvas_vector.data();
53 }
54 std::fill(keydown, keydown + 128, false);
55}

57void Window::close() {
58 opening = false;
59 if (opengl) {
60 SDL_GL_DeleteContext(context);
61 } else {
62 SDL_FreeSurface(surface);
63 }
64 SDL_DestroyWindow(sdl_window);
65 SDL_Quit();
66}

68void Window::update() {
69 /* init key pressed & released array */
70 std::fill(keypressed, keypressed + 512, false);
71 std::fill(keyreleased, keyreleased + 512, false);
72 /* deal with events, low performance on Mac or Linux */
73 SDL_Event event;
74 while (SDL_PollEvent(&event)) {
75 int32_t keycode = event.key.keysym.sym;
76 if (event.type == SDL_QUIT) {
77 close();
78 return;
79 } else if (event.type == SDL_KEYDOWN) {
80 if (keycode < 512) {
81 keypressed[keycode] = !keydown[keycode];
82 keydown[keycode] = true;
83 }
84 } else if (event.type == SDL_KEYUP) {
85 if (keycode < 512) {
86 keyreleased[keycode] = true;
87 keydown[keycode] = false;
88 }
89 } else if (event.type == SDL_MOUSEBUTTONDOWN) {
90 if (event.button.button == SDL_BUTTON_LEFT) {
91 keypressed[1] = true;
92 keydown[1] = true;
93 } else if (event.button.button == SDL_BUTTON_RIGHT) {
94 keypressed[2] = true;
95 keydown[2] = true;
96 }
97 } else if (event.type == SDL_MOUSEBUTTONUP) {
98 if (event.button.button == SDL_BUTTON_LEFT) {
99 keypressed[1] = true;
100 keydown[1] = false;
101 } else if (event.button.button == SDL_BUTTON_RIGHT) {
102 keypressed[2] = true;
103 keydown[2] = false;
104 }
105 } else if (event.type == SDL_MOUSEMOTION) {
106 cursorx = event.motion.x;
107 cursory = event.motion.y;
108 }
109 }
110 /* lock cursor to the center of window */
111 if (lock_cursor && SDL_GetKeyboardFocus()) {
112 SDL_WarpMouseInWindow(sdl_window, width / 2, height / 2);
113 }
114 /* refresh window */
115 if (opengl) {
116 SDL_GL_SwapWindow(sdl_window);
117 } else {
118 std::copy(canvas_vector.begin(), canvas_vector.end(), static_cast<uint32_t*>(surface->pixels));
119 SDL_UpdateWindowSurface(sdl_window);
120 }
121 /* calcuate delta time */
122 deltatime = SDL_GetTicks() - time;
123 if (deltatime <= interval) SDL_Delay(interval - deltatime);
124 time = SDL_GetTicks();
125 deltatime = std::max(interval, deltatime);
126}
