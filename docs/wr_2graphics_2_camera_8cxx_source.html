35void Camera::persp(float f, float a, float zn, float zf) {
36 fovy = f;
37 aspect = a;
38 znear = zn;
39 zfar = zf;
40 projection = {
41 1 / tanf(f / 2) / a, 0, 0, 0,
42 0, 1 / tanf(f / 2), 0, 0,
43 0, 0, (zn + zf) / (zn - zf), 2 * zf * zn / (zn - zf),
44 0, 0, -1, 0,
45 };
46}

48void Camera::ortho(float l, float r, float low, float up, float zf, float zn) {
49 left = l;
50 right = r;
51 lower = low;
52 upper = up;
53 zfar = zf;
54 znear = zn;
55 projection = {
56 2 / (r - l), 0, 0, (r + l) / (l - r),
57 0, 2 / (up - low), 0, (up + low) / (low - up),
58 0, 0, 2 / (zn - zf), (zf + zn) / (zn - zf),
59 0, 0, 0, 1,
60 };
61}

63void Camera::lookat(const Vec3& p, const Vec3& d, const Vec3& u) {
64 position = p;
65 direction = d;
66 up = u;
67 Vec3 r = (-d) ^u;
68 viewing = {
69 r.x, r.y, r.z, -(p * r),
70 u.x, u.y, u.z, -(p * u),
71 d.x, d.y, d.z, -(p * d),
72 0, 0, 0, 1,
73 };
74}

76void Camera::model_transform(const Vec3& p, const Vec3& r, const Vec3& s,
77 Mat<4, 4>& matt, Mat<4, 4>& matr, Mat<4, 4>& mats) {
78 matt = { /* translation matrix */
79 1, 0, 0, p.x,
80 0, 1, 0, p.y,
81 0, 0, 1, p.z,
82 0, 0, 0, 1,
83 };
84 matr = /* rotation matrices */
85 Mat<4, 4>{
86 1, 0, 0, 0, /* rotation x matrix */
87 0, cosf(r.x), -sinf(r.x), 0,
88 0, sinf(r.x), cosf(r.x), 0,
89 0, 0, 0, 1,
90 } *
91 Mat<4, 4>{
92 cosf(r.y), 0, -sinf(r.y), 0, /* rotation y matrix */
93 0, 1, 0, 0,
94 sinf(r.y), 0, cosf(r.y), 0,
95 0, 0, 0, 1,
96 } *
97 Mat<4, 4>{
98 cosf(r.z), -sinf(r.z), 0, 0, /* rotation z matrix */
99 sinf(r.z), cosf(r.z), 0, 0,
100 0, 0, 1, 0,
101 0, 0, 0, 1,
102 };
103 mats = { /* scaling matrix */
104 s.x, 0, 0, 0,
105 0, s.y, 0, 0,
106 0, 0, s.z, 0,
107 0, 0, 0, 1,
108 };
109}
