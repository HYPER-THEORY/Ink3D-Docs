197void Scene::update_instances() {
198 std::vector<Instance*> unvisited;
199 matrix_local = transform();
200 matrix_global = matrix_local;
201 unvisited.emplace_back(this);
202 while (!unvisited.empty()) {
203 Instance* current = unvisited.back();
204 unvisited.pop_back();
205 size_t count = current->get_child_count();
206 for (int i = 0; i < count; ++i) {
207 Instance* child = current->get_child(i);
208 child->matrix_local = child->transform();
209 child->matrix_global = current->matrix_global * child->matrix_local;
210 unvisited.emplace_back(child);
211 }
212 }
213}

231std::vector<const Instance*> Scene::to_visible_instances() const {
232 std::vector<const Instance*> unvisited;
233 std::vector<const Instance*> instances;
234 unvisited.emplace_back(this);
235 while (!unvisited.empty()) {
236 const Instance* current = unvisited.back();
237 unvisited.pop_back();
238 size_t count = current->get_child_count();
239 for (int i = 0; i < count; ++i) {
240 unvisited.emplace_back(current->get_child(i));
241 }
242 if (current->visible && current->mesh != nullptr) {
243 instances.emplace_back(current);
244 }
245 }
246 return instances;
