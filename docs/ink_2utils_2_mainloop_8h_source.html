56int run() {
57 Settings t; /* create settings */
58 conf(t); /* config settings */
59
60 /* initialize window with parameters */
61 Ink::Window::init(t.title, t.x, t.y, t.width, t.height, t.highdpi);
62 Ink::Window::init_opengl(t.vsync, t.depth, t.stencil, t.msaa);
63 Ink::Window::set_fullscreen(t.fullscreen);
64 Ink::Window::set_borderless(t.borderless);
65 Ink::Window::set_resizable(t.resizable);
66 Ink::Window::set_min_size(t.min_width, t.min_height);
67 Ink::Window::set_interval(1000 / t.fps);
68 Ink::Window::set_cursor_visible(t.show_cursor);
69 Ink::Window::set_cursor_locked(t.lock_cursor);
70 if (t.lock_cursor) Ink::Window::set_cursor_position(t.width / 2, t.height / 2);
71
72 /* initialize the viewport of render pass */
73 int width = t.width * (t.highdpi ? 2 : 1);
74 int height = t.height * (t.highdpi ? 2 : 1);
75 Ink::RenderPass::set_viewport(Ink::Gpu::Rect(width, height));
76
77 /* print error from Ink3D */
78 Ink::Error::set_callback([](const std::string& s) -> void {
79 std::cerr << Ink::Date::get_local().format() << ' ' << s << '\n';
80 });
81
82 load(); /* load once */
83
84 /* main loop */
85 uint32_t time = 0;
86 while (Ink::Window::is_open()) {
87
88 /* clear default frame buffer */
89 Ink::Gpu::FrameBuffer::activate(nullptr);
90 Ink::Gpu::set_clear_color(t.background_color, 0);
91 Ink::Gpu::clear();
92
93 /* update window with delta time */
94 uint32_t delta_time = Ink::Window::get_time() - time;
95 time = Ink::Window::get_time();
96 update(delta_time * 0.001f);
97 Ink::Window::update();
98
99 /* print error from OpenGL */
100 std::string error = Ink::Gpu::get_error();
101 if (!error.empty()) std::cerr << error;
102
103#ifdef DEBUG
104 /* close window when pressing escape key */
105 if (Ink::Window::is_down(SDLK_ESCAPE)) {
106 Ink::Window::close();
107 }
108#endif
109 }
110
111 quit(); /* quit window */
112
113 return 0;
114}
