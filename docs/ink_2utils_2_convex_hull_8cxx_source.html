51void ConvexHull::compute() {
52 insert_face(0, 1, 2);
53 insert_face(2, 1, 0);
54 size_t size = vertices.size();
55 for (int i = 3; i < size; ++i) {
56 std::unordered_set<long long> new_faces;
57 auto face_iter = faces.begin();
58 auto normal_iter = normals.begin();
59 while (face_iter != faces.end()) {
60 Vec3 dir = vertices[i] - vertices[(*face_iter)[0]];
61 if (normal_iter->dot(dir) <= 0) {
62 ++face_iter;
63 ++normal_iter;
64 continue;
65 }
66 for (int k = 0; k < 3; ++k) {
67 long long u = (*face_iter)[k];
68 long long v = (*face_iter)[(k + 1) % 3];
69 long long uv = v << 32 | u;
70 if (new_faces.count(uv) != 0) {
71 new_faces.erase(uv);
72 } else {
73 new_faces.insert(u << 32 | v);
74 }
75 }
76 face_iter = faces.erase(face_iter);
77 normal_iter = normals.erase(normal_iter);
78 }
79 for (auto& f : new_faces) {
80 insert_face(i, f >> 32, f & 0xFFFFFFFFll);
81 }
82 }
83}
