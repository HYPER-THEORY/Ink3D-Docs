64void IBLFilter::load_texture(const Gpu::Texture& t, Gpu::Texture& m, int s) {
65 /* initialize load texture */
66 [[maybe_unused]]
67 bool inited = init_load_texture();
68
69 /* disable depth & stencil & scissor test */
70 Gpu::disable_depth_test();
71 Gpu::disable_stencil_test();
72 Gpu::disable_scissor_test();
73
74 /* disable blending & wireframe & culling */
75 Gpu::disable_blending();
76 Gpu::disable_wireframe();
77 Gpu::disable_culling();
78
79 /* enable texture cube seamless */
80 Gpu::enable_texture_cube_seamless();
81
82 /* change the current viewport */
83 Gpu::set_viewport(Gpu::Rect(s, s));
84
85 /* initialize radiance environment map */
86 m.init_cube(s, s, t.get_format());
87 m.set_filters(TEXTURE_LINEAR, TEXTURE_LINEAR_MIPMAP_LINEAR);
88 m.generate_mipmap();
89
90 /* fetch the blur shader from shader lib */
91 auto* blur_shader = ShaderLib::fetch("SphericalBlur");
92
93 /* fetch the cubemap shader from shader lib */
94 Defines cubemap_defines;
95 int type = t.get_type();
96 cubemap_defines.set_if("USE_EQUIRECT", type == TEXTURE_2D);
97 cubemap_defines.set_if("USE_CUBEMAP", type == TEXTURE_CUBE);
98 auto* cubemap_shader = ShaderLib::fetch("Cubemap", cubemap_defines);
99
100 /* prepare cubemap frame buffer */
101 if (!cubemap_target) {
102 cubemap_target = std::make_unique<Gpu::FrameBuffer>();
103 cubemap_target->draw_attachments({0});
104 }
105
106 /* render to cube texture */
107 for (int i = 0; i < 6; ++i) {
108 cubemap_target->set_attachment(m, 0, 0, i);
109 Gpu::FrameBuffer::activate(cubemap_target.get());
110 cubemap_shader->use_program();
111 cubemap_shader->set_uniform_i("face", i);
112 cubemap_shader->set_uniform_i("map", t.activate(0));
113 plane->attach(*cubemap_shader);
114 plane->draw();
115 }
116
117 /* prepare blur map */
118 if (!blur_map) blur_map = std::make_unique<Gpu::Texture>();
119 blur_map->init_cube(s, s, t.get_format());
120 blur_map->generate_mipmap();
121 blur_map->set_filters(TEXTURE_LINEAR, TEXTURE_LINEAR_MIPMAP_LINEAR);
122
123 /* prepare blur frame buffer */
124 if (!blur_target) {
125 blur_target = std::make_unique<Gpu::FrameBuffer>();
126 blur_target->draw_attachments({0});
127 }
128
129 /* blur cube texture latitudinally and longitudinally */
130 float weights[20];
131 int size_lod = s;
132 int max_lod = log2f(size_lod);
133 for (int lod = 1; lod <= max_lod; ++lod) {
134
135 /* calculate sigma radians */
136 float sigma_radians = sqrtf(3) / size_lod;
137 if (lod == 1) sigma_radians = 2.f / size_lod;
138
139 /* calculate blur parameters */
140 Vec3 pole_axis = AXES[(lod - 1) % 10];
141 float d_theta = PI / (size_lod * 2 - 2);
142 float sigma = sigma_radians / d_theta;
143 int samples = 1 + floorf(sigma * 3);
144 gaussian_weights(sigma, 20, weights);
145
146 /* change the current viewport */
147 Gpu::set_viewport(Gpu::Rect(size_lod / 2, size_lod / 2));
148
149 /* blur texture latitudinally */
150 for (int i = 0; i < 6; ++i) {
151 blur_target->set_attachment(*blur_map, 0, lod, i);
152 Gpu::FrameBuffer::activate(blur_target.get());
153 blur_shader->use_program();
154 blur_shader->set_uniform_i("face", i);
155 blur_shader->set_uniform_i("samples", samples);
156 blur_shader->set_uniform_i("latitudinal", 1);
157 blur_shader->set_uniform_f("d_theta", d_theta);
158 blur_shader->set_uniform_f("lod", lod - 1);
159 blur_shader->set_uniform_v3("pole_axis", pole_axis);
160 blur_shader->set_uniform_i("map", m.activate(0));
161 for (int w = 0; w < 20; ++w) {
162 std::string weights_i = Format::format("weights[{}]", w);
163 blur_shader->set_uniform_f(weights_i, weights[w]);
164 }
165 plane->attach(*blur_shader);
166 plane->draw();
167 }
168
169 /* calculate blur parameters */
170 size_lod /= 2;
171 d_theta = PI / (size_lod * 2 - 2);
172 if (std::isinf(d_theta)) d_theta = PI;
173 sigma = sigma_radians / d_theta;
174 samples = 1 + floor(sigma * 3);
175 gaussian_weights(sigma, 20, weights);
176
177 /* blur texture longitudinally */
178 for (int f = 0; f < 6; ++f) {
179 blur_target->set_attachment(m, 0, lod, f);
180 Gpu::FrameBuffer::activate(blur_target.get());
181 blur_shader->use_program();
182 blur_shader->set_uniform_i("face", f);
183 blur_shader->set_uniform_i("samples", samples);
184 blur_shader->set_uniform_i("latitudinal", 0);
185 blur_shader->set_uniform_f("d_theta", d_theta);
186 blur_shader->set_uniform_f("lod", lod);
187 blur_shader->set_uniform_v3("pole_axis", pole_axis);
188 blur_shader->set_uniform_i("map", blur_map->activate(0));
189 for (int w = 0; w < 20; ++w) {
190 std::string weights_i = Format::format("weights[{}]", w);
191 blur_shader->set_uniform_f(weights_i, weights[w]);
192 }
193 plane->attach(*blur_shader);
194 plane->draw();
195 }
196 }
197
198 /* set back to the initial viewport */
199 Gpu::FrameBuffer::activate(nullptr);
200}

202bool IBLFilter::init_load_texture() {
203 /* prepare plane mesh */
204 Mesh plane_mesh = Mesh();
205 plane_mesh.groups = {{"default", 0, 3}};
206 plane_mesh.vertex = {{-1, 3, 0}, {-1, -1, 0}, {3, -1, 0}};
207 plane_mesh.uv = {{0, 2}, {0, 0}, {2, 0}};
208
209 /* prepare plane vertex object */
210 plane = std::make_unique<Gpu::VertexObject>();
211 plane->load(plane_mesh);
212
213 return true; /* finish */
214}

216void IBLFilter::gaussian_weights(float s, int n, float* w) {
217 float weight_sum = 0;
218 for (int i = 0; i < n; ++i) {
219 float x = i / s;
220 float weight = expf(-x * x * .5f);
221 w[i] = weight;
222 if (i == 0) {
223 weight_sum += weight;
224 } else {
225 weight_sum += weight * 2;
226 }
227 }
228 for (int i = 0; i < 20; ++i) {
229 w[i] /= weight_sum;
230 }
231}
