83void Solid::collide(Solid* s, float x, float y, float z) {
84 CollisionBox& collided = s->box;
85 if (s != this && box.hittest(collided)) {
86 if (x > 0) {
87 position.x = collided.v1.x - width - limit;
88 } else if (x < 0) {
89 position.x = collided.v2.x + limit;
90 }
91 if (y > 0) {
92 position.y = collided.v1.y - height - limit;
93 } else if (y < 0) {
94 position.y = collided.v2.y + limit;
95 }
96 if (z > 0) {
97 position.z = collided.v1.z - depth - limit;
98 } else if (z < 0) {
99 position.z = collided.v2.z + limit;
100 }
101 refresh();
102 }
103}
