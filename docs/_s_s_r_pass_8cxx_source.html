30void SSRPass::init() {
31 /* calculate max lod and Hierarchical-Z map size */
32 Vec2 max_lod = {ceilf(log2f(width)) - 1, ceilf(log2f(height)) - 1};
33 Vec2 z_map_size = {powf(2, max_lod.x), powf(2, max_lod.y)};
34
35 /* prepare Hierarchical-Z map */
36 z_map = std::make_unique<Gpu::Texture>();
37 z_map->init_2d(z_map_size.x, z_map_size.y, TEXTURE_R32_SFLOAT);
38 z_map->set_filters(TEXTURE_NEAREST, TEXTURE_NEAREST_MIPMAP_NEAREST);
39 z_map->generate_mipmap();
40
41 /* prepare Hierarchical-Z frame buffer */
42 hi_z_target = std::make_unique<Gpu::FrameBuffer>();
43 hi_z_target->draw_attachments({0});

46void SSRPass::render() const {
47 /* fetch copy shader from shader lib */
48 auto* copy_shader = ShaderLib::fetch("Copy");
49
50 /* fetch Hierarchical-Z shader from shader lib */
51 auto* hi_z_shader = ShaderLib::fetch("MinHiZ");
52
53 /* fetch SSR shader from shader lib */
54 auto* ssr_shader = ShaderLib::fetch("SSR");
55
56 /* calculate max lod and Hierarchical-Z map size */
57 Vec2 max_lod = {ceilf(log2f(width)) - 1, ceilf(log2f(height)) - 1};
58 Vec2 z_map_size = {powf(2, max_lod.x), powf(2, max_lod.y)};
59
60 /* the max level used in algorithm */
61 int max_level = fmin(max_lod.x, max_lod.y);
62
63 /* change the current viewport */
64 Gpu::Rect viewport = RenderPass::get_viewport();
65 RenderPass::set_viewport(Gpu::Rect(z_map_size.x, z_map_size.y));
66
67 /* 1. copy depth buffer to Hierarchical-Z map */
68 copy_shader->use_program();
69 copy_shader->set_uniform_i("map", buffer_d->activate(0));
70 hi_z_target->set_attachment(*z_map, 0, 0);
71 RenderPass::render_to(copy_shader, hi_z_target.get());
72
73 /* 2. generate Hierarchical-Z map */
74 Vec2 z_size = z_map_size / 2;
75 for (int lod = 0; lod < max_level; ++lod) {
76
77 /* set the viewport to Hierarchical-Z map size */
78 RenderPass::set_viewport(Gpu::Rect(z_size.x, z_size.y));
79
80 /* set the mipmap range of Hierarchical-Z map */
81 /* to avoid reading and writing with the same texture */
82 z_map->set_mipmap_range(lod, lod);
83
84 /* render to lower level Hierarchical-Z map */
85 hi_z_shader->use_program();
86 hi_z_shader->set_uniform_v2("screen_size", z_size);
87 hi_z_shader->set_uniform_i("lod", lod);
88 hi_z_shader->set_uniform_i("map", z_map->activate(0));
89 hi_z_target->set_attachment(*z_map, 0, lod + 1);
90 RenderPass::render_to(hi_z_shader, hi_z_target.get());
91
92 /* update the size of Hierarchical-Z map */
93 z_size.x = fmax(1, floorf(z_size.x / 2));
94 z_size.y = fmax(1, floorf(z_size.y / 2));
95 }
96
97 /* set back to initial the mipmap range */
98 z_map->set_mipmap_range(0, 1000);
99
100 /* set back to the initial viewport */
101 RenderPass::set_viewport(viewport);
102
103 /* calculate camera parameters */
104 Mat4 view_proj = camera->projection * camera->viewing;
105 Mat4 inv_view_proj = inverse_4x4(view_proj);
106
107 /* 3. render SSR mixing with input map to render target */
108 ssr_shader->use_program();
109 ssr_shader->set_uniform_i("max_steps", max_steps);
110 ssr_shader->set_uniform_i("max_level", max_level);
111 ssr_shader->set_uniform_f("intensity", intensity);
112 ssr_shader->set_uniform_f("thickness", thickness);
113 ssr_shader->set_uniform_f("max_roughness", max_roughness);
114 ssr_shader->set_uniform_v3("camera_pos", camera->position);
115 ssr_shader->set_uniform_f("camera_near", camera->near);
116 ssr_shader->set_uniform_f("camera_far", camera->far);
117 ssr_shader->set_uniform_v2("z_map_size", z_map_size);
118 ssr_shader->set_uniform_m4("view_proj", view_proj);
119 ssr_shader->set_uniform_m4("inv_view_proj", inv_view_proj);
120 ssr_shader->set_uniform_i("map", map->activate(0));
121 ssr_shader->set_uniform_i("z_map", z_map->activate(1));
122 ssr_shader->set_uniform_i("buffer_n", buffer_n->activate(2));
123 ssr_shader->set_uniform_i("buffer_m", buffer_n->activate(3));
124 RenderPass::render_to(ssr_shader, target);
125}
