37float nearest_map(const Image& t, int c, float u, float v) {
38 int x = roundf(u * (t.width - 1));
39 x = x < 0 ? 0 : x;
40 x = x >= t.width ? t.width - 1 : x;
41 int y = roundf(v * (t.height - 1));
42 y = y < 0 ? 0 : y;
43 y = y >= t.height ? t.height - 1 : y;
44 int bpp = t.channel * t.bytes;
45 auto* ptr = t.data.data() + (x + y * t.width) * bpp;
46 return ptr[c] / 255.f;
47}

53float linear_map(const Image& t, int c, float u, float v) {
54 int x0 = u * (t.width - 1);
55 x0 = x0 < 0 ? 0 : x0;
56 x0 = x0 >= t.width ? t.width - 1 : x0;
57 int x1 = x0 + 1;
58 x1 = x1 >= t.width ? t.width - 1 : x1;
59 int y0 = v * (t.height - 1);
60 y0 = y0 < 0 ? 0 : y0;
61 y0 = y0 >= t.height ? t.height - 1 : y0;
62 int y1 = y0 + 1;
63 y1 = y1 >= t.height ? t.height - 1 : y1;
64 int bpp = t.channel * t.bytes;
65 auto* data = t.data.data();
66 auto* ptr_0 = data + (x0 + y0 * t.width) * bpp;
67 auto* ptr_1 = data + (x0 + y1 * t.width) * bpp;
68 auto* ptr_2 = data + (x1 + y0 * t.width) * bpp;
69 auto* ptr_3 = data + (x1 + y1 * t.width) * bpp;
70 return (ptr_0[c] * (y1 - v) + ptr_1[c] * (v - y0)) * (x1 - u) +
71 (ptr_2[c] * (y1 - v) + ptr_3[c] * (v - y0)) * (u - x0);
72}

89void znear_clip(const PointList& i, float z, PointList& o) {
90 for (int l = 0; l < i.size; ++l) {
91 Vec4& vertex1 = i.vertices[l];
92 Vec4& vertex2 = i.vertices[(l + 1) % i.size];
93
94 /* vertices are both outside */
95 if (vertex1.w < z && vertex2.w < z) continue;
96
97 /* vertices are both inside */
98 const Vec3& barycenter1 = i.barycenters[l];
99 const Vec3& barycenter2 = i.barycenters[(l + 1) % i.size];
100 if (vertex1.w > z && vertex2.w > z) {
101 add_point(o, vertex2, barycenter2);
102 continue;
103 }
104
105 /* traveling from inside to outside */
106 float weight1 = fabsf(vertex1.w - z);
107 float weight2 = fabsf(vertex2.w - z);
108 float inverse = 1 / (weight1 + weight2);
109 if (vertex1.w > z && vertex2.w < z) {
110 add_point(o, (vertex1 * weight2 + vertex2 * weight1) * inverse,
111 (barycenter1 * weight2 + barycenter2 * weight1) * inverse);
112 continue;
113 }
114
115 /* traveling from outside to inside */
116 add_point(o, (vertex1 * weight2 + vertex2 * weight1) * inverse,
117 (barycenter1 * weight2 + barycenter2 * weight1) * inverse);
118 add_point(o, vertex2, barycenter2);
119 }

155void rasterize(const PointList& p, const Vec3* d, Shader& s, double* zb, Vec4* canvas) {
156 for (int i = 2; i < p.size; ++i) {
157 const Vec3& vertex_a = d[0];
158 const Vec3& vertex_b = d[i - 1];
159 const Vec3& vertex_c = d[i];
160 Vec3 fixed_a = {p.barycenters[0].x, p.barycenters[0].y, p.barycenters[0].z};
161 Vec3 fixed_b = {p.barycenters[i - 1].x, p.barycenters[i - 1].y, p.barycenters[i - 1].z};
162 Vec3 fixed_c = {p.barycenters[i].x, p.barycenters[i].y, p.barycenters[i].z};
163 Vec3 fixed_barycenter = {1 / p.vertices[0].w, 1 / p.vertices[i - 1].w, 1 / p.vertices[i].w};
164 DVec2 v0 = {vertex_c.x - vertex_a.x, vertex_c.y - vertex_a.y};
165 DVec2 v1 = {vertex_b.x - vertex_a.x, vertex_b.y - vertex_a.y};
166 DVec2 v2 = {};
167 double dot00 = v0.dot(v0);
168 double dot01 = v0.dot(v1);
169 double dot11 = v1.dot(v1);
170 double inverse = 1 / (dot00 * dot11 - dot01 * dot01);
171 Vec3 barycenter;
172 Vec3 vertex_l = vertex_a;
173 Vec3 vertex_m = vertex_b;
174 Vec3 vertex_u = vertex_c;
175 if (vertex_l.y > vertex_m.y) std::swap(vertex_l, vertex_m);
176 if (vertex_m.y > vertex_u.y) std::swap(vertex_m, vertex_u);
177 if (vertex_l.y > vertex_m.y) std::swap(vertex_l, vertex_m);
178 float lower = fmaxf(floorf(vertex_l.y) + 1, 0);
179 float upper = fminf(floorf(vertex_u.y) + 1, viewport_h);
180 float median = vertex_m.y;
181 float inverse_ml = 1 / (vertex_m.y - vertex_l.y);
182 float inverse_um = 1 / (vertex_u.y - vertex_m.y);
183 float inverse_ul = 1 / (vertex_u.y - vertex_l.y);
184 for (float y = lower; y < upper; ++y) {
185 float left = y < median ?
186 (vertex_l.x * (vertex_m.y - y) + vertex_m.x * (y - vertex_l.y)) * inverse_ml :
187 (vertex_m.x * (vertex_u.y - y) + vertex_u.x * (y - vertex_m.y)) * inverse_um;
188 float right = (vertex_l.x * (vertex_u.y - y) + vertex_u.x * (y - vertex_l.y)) * inverse_ul;
189 if (left > right) std::swap(left, right);
190 left = fmaxf(floorf(left) + 1, 0);
191 right = fminf(floorf(right) + 1, viewport_w);
192 for (float x = left; x < right; ++x) {
193 v2.x = x - vertex_a.x;
194 v2.y = y - vertex_a.y;
195 double dot02 = v0.dot(v2);
196 double dot12 = v1.dot(v2);
197 double u = (dot11 * dot02 - dot01 * dot12) * inverse;
198 double v = (dot00 * dot12 - dot01 * dot02) * inverse;
199 if (UNLIKELY(u < -EPS || v < -EPS || u + v > 1 + EPS)) continue;
200 double z = vertex_a.z * (1 - u - v) + vertex_b.z * v + vertex_c.z * u;
201 int location = x + y * viewport_w;
202 if (z > -1 && z < 1 && z < zb[location] + EPS) {
203 zb[location] = z;
204 barycenter.x = (1 - u - v) * fixed_barycenter.x;
205 barycenter.y = v * fixed_barycenter.y;
206 barycenter.z = u * fixed_barycenter.z;
207 barycenter /= barycenter.x + barycenter.y + barycenter.z;
208 barycenter = fixed_a * barycenter.x + fixed_b * barycenter.y + fixed_c * barycenter.z;
209 s.fragment(barycenter, {x / viewport_w, y / viewport_h}, canvas[location]);
210 }
211 }
212 }

264void draw(const Camera& c, Shader& s, const Mesh& m, double* zb, Vec4* canvas) {
265 Vec3 barycenters[3] = {{1, 0, 0}, {0, 1, 0}, {0, 0, 1}};
266 size_t length = m.vertex.size();
267 for (int i = 0; i < length; i += 3) {
268
269 /* vertex shader & geometry shader */
270 Vec4 vertices[3];
271 for (int j = 0; j < 3; ++j) {
272 s.vextex(m, i + j, j, vertices[j]);
273 }
274 s.geometry(vertices);
275
276 /* do Z-near clipping */
277 Vec4 clip_vertices[4];
278 Vec3 clip_barycenters[4];
279 PointList clip_primitive = {0, clip_vertices, clip_barycenters};
280 znear_clip({3, vertices, barycenters}, c.near, clip_primitive);
281
282 /* do Z-far clipping */
283 Vec4 primitive_vertices[5];
284 Vec3 primitive_barycenters[5];
285 PointList primitive = {0, primitive_vertices, primitive_barycenters};
286 zfar_clip(clip_primitive, c.far, primitive);
287
288 /* perspective division */
289 Vec3 device_vertices[5];
290 for (int i = 0; i < primitive.size; ++i) {
291 device_vertices[i] = Vec3(
292 primitive.vertices[i].x,
293 primitive.vertices[i].y,
294 primitive.vertices[i].z
295 ) / primitive.vertices[i].w;
296 }
297
298 /* viewport transform */
299 int width_2 = viewport_w / 2;
300 int height_2 = viewport_h / 2;
301 for (int i = 0; i < primitive.size; ++i) {
302 device_vertices[i].x = device_vertices[i].x * width_2 + width_2;
303 device_vertices[i].y = -device_vertices[i].y * height_2 + height_2;
304 }
305
306 /* rasterization */
307 if (canvas == nullptr) {
308 return rasterize(primitive, device_vertices, zb);
309 }
310 rasterize(primitive, device_vertices, s, zb, canvas);
311 }
