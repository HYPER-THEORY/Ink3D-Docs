32void Window::init(const std::string& t, int x, int y, int w, int h, bool opengl, bool dpi) {
33 /* whether to center the window */
34 if (x == -1) x = SDL_WINDOWPOS_CENTERED;
35 if (y == -1) y = SDL_WINDOWPOS_CENTERED;
36
37 /* initialize keydown with false */
38 std::fill_n(keydown, 512, false);
39
40 /* check whether to enable OpenGL */
41 Window::opengl = opengl;
42
43 /* create a SDL window */
44 SDL_Init(SDL_INIT_VIDEO);
45 uint32_t flags = SDL_WINDOW_SHOWN;
46 flags |= SDL_WINDOW_OPENGL * opengl;
47 flags |= SDL_WINDOW_ALLOW_HIGHDPI * dpi;
48 sdl_window = SDL_CreateWindow(t.c_str(), x, y, w, h, flags);
49}

56void Window::init_opengl(int v, int d, int s, int m, bool a) {
57 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
58 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 4);
59 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 1);
60 SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, d);
61 SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, s);
62 SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
63 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, m != 0);
64 SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, m);
65 SDL_GL_SetAttribute(SDL_GL_ACCELERATED_VISUAL, a);
66 context = SDL_GL_CreateContext(sdl_window);
67 SDL_GL_SetSwapInterval(v);
68 if (gladLoadGL() == 0) Error::set("Window: Failed to load OpenGL");
69}

78void Window::update() {
79 /* init key pressed & released array */
80 std::fill_n(keypressed, 512, false);
81 std::fill_n(keyreleased, 512, false);
82
83 /* handle events */
84 SDL_Event event;
85 while (SDL_PollEvent(&event)) {
86 int32_t keycode = event.key.keysym.sym;
87 if (event.type == SDL_QUIT) {
88 close();
89 return;
90 } else if (event.type == SDL_KEYDOWN) {
91 if (keycode < 512) {
92 keypressed[keycode] = !keydown[keycode];
93 keydown[keycode] = true;
94 }
95 } else if (event.type == SDL_KEYUP) {
96 if (keycode < 512) {
97 keyreleased[keycode] = true;
98 keydown[keycode] = false;
99 }
100 } else if (event.type == SDL_MOUSEBUTTONDOWN) {
101 if (event.button.button == SDL_BUTTON_LEFT) {
102 keypressed[1] = true;
103 keydown[1] = true;
104 } else if (event.button.button == SDL_BUTTON_RIGHT) {
105 keypressed[2] = true;
106 keydown[2] = true;
107 }
108 } else if (event.type == SDL_MOUSEBUTTONUP) {
109 if (event.button.button == SDL_BUTTON_LEFT) {
110 keypressed[1] = true;
111 keydown[1] = false;
112 } else if (event.button.button == SDL_BUTTON_RIGHT) {
113 keypressed[2] = true;
114 keydown[2] = false;
115 }
116 } else if (event.type == SDL_MOUSEMOTION) {
117 cursor_x = event.motion.x;
118 cursor_y = event.motion.y;
119 }
120 }
121
122 /* lock cursor to the center of window */
123 if (cursor_locked && SDL_GetKeyboardFocus()) {
124 int dimensions[2];
125 SDL_GetWindowSize(sdl_window, &dimensions[0], &dimensions[1]);
126 SDL_WarpMouseInWindow(sdl_window, dimensions[0] / 2, dimensions[1] / 2);
127 }
128
129 /* refresh window */
130 if (opengl) {
131 SDL_GL_SwapWindow(sdl_window);
132 } else {
133 SDL_UpdateWindowSurface(sdl_window);
134 }
135
136 /* calculate delta time */
137 deltatime = SDL_GetTicks() - time;
138 if (deltatime <= interval) SDL_Delay(interval - deltatime);
139 time = SDL_GetTicks();
140 deltatime = std::max(interval, deltatime);
141}
