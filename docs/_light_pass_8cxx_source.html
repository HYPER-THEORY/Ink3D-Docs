31void LightPass::render() const {
32 /* fetch light shader from shader lib */
33 Defines light_defines = Renderer::define_scene(*scene);
34 light_defines.set(Renderer::define_tone_map(tone_map_mode));
35 auto* light_shader = ShaderLib::fetch("Lighting", light_defines);
36
37 /* pass parameters and G-Buffers to shader */
38 light_shader->use_program();
39 light_shader->set_uniform_v3("camera_pos", camera->position);
40 light_shader->set_uniform_f("exposure", tone_map_exposure);
41 light_shader->set_uniform_i("buffer_c", buffer_c->activate(0));
42 light_shader->set_uniform_i("buffer_n", buffer_n->activate(1));
43 light_shader->set_uniform_i("buffer_m", buffer_m->activate(2));
44 light_shader->set_uniform_i("buffer_a", buffer_a->activate(3));
45 light_shader->set_uniform_i("buffer_d", buffer_d->activate(4));
46
47 /* pass camera parameter to shader */
48 Mat4 inv_view_proj = inverse_4x4(camera->projection * camera->viewing);
49 light_shader->set_uniform_m4("inv_view_proj", inv_view_proj);
50
51 /* pass the lights & fogs parameters to shader */
52 Renderer::set_light_uniforms(*scene, *light_shader);
53
54 /* render results to render target */
55 RenderPass::render_to(light_shader, target);
56}
