35Image Loader::load_image(const std::string& p) {
36 /* create a new image */
37 Image image = Image(0, 0, 0, 1);
38
39 /* get image data from file */
40 auto* image_ptr = stbi_load(p.c_str(), &image.width, &image.height, &image.channel, 0);
41 if (image_ptr == nullptr) {
42 Error::set("Loader: Error reading from image");
43 return Image();
44 }
45
46 /* copy image data to new image */
47 image.data.resize(image.width * image.height * image.channel * image.bytes);
48 memcpy(image.data.data(), image_ptr, image.data.size());
49
50 /* release image data */
51 stbi_image_free(image_ptr);
52 return image;

75std::vector<Mesh> Loader::load_obj(const std::string& p, const std::string& g) {
76 /* prepare the file stream */
77 std::ifstream stream;
78 stream.open(p, std::ifstream::in);
79 if (!stream) {
80 Error::set("Loader: Error reading from obj file");
81 return std::vector<Mesh>();
82 }
83 size_t stream_max = std::numeric_limits<std::streamsize>::max();
84
85 /* temporary data */
86 std::vector<Vec3> vertex;
87 std::vector<Vec3> color;
88 std::vector<Vec3> normal;
89 std::vector<Vec2> uv;
90
91 /* initialize mesh vector */
92 std::vector<Mesh> meshes;
93 meshes.emplace_back(Mesh("default"));
94 Mesh* current_mesh = &meshes.back();
95
96 /* initialize mesh group pointer */
97 current_mesh->groups.emplace_back<MeshGroup>({"default", 0, 0});
98 MeshGroup* current_group = &current_mesh->groups.back();
99
100 /* initialize total length */
101 int total_length = 0;
102
103 /* read data by line */
104 while (!stream.eof()) {
105 std::string keyword;
106 stream >> keyword;
107
108 /* add vertex to temporary array */
109 if (keyword == "v") {
110 Vec3 v;
111 stream >> v.x >> v.y >> v.z;
112 vertex.emplace_back(v);
113 }
114
115 /* add normal to temporary array */
116 else if (keyword == "vn") {
117 Vec3 vn;
118 stream >> vn.x >> vn.y >> vn.z;
119 normal.emplace_back(vn);
120 }
121
122 /* add uv to temporary array */
123 else if (keyword == "vt") {
124 Vec2 vt;
125 stream >> vt.x >> vt.y;
126 uv.emplace_back(vt);
127 }
128
129 /* search for data by index and add it to current mesh */
130 else if (keyword == "f") {
131 int index = 0;
132 for (int i = 0; i < 3; ++i) {
133 stream >> index;
134 current_mesh->vertex.emplace_back(vertex[index - 1]);
135 stream.get();
136
137 /* check whether uv is omitted */
138 if (std::isdigit(stream.peek())) {
139 stream >> index;
140 current_mesh->uv.emplace_back(uv[index - 1]);
141 }
142
143 stream.get();
144 stream >> index;
145 current_mesh->normal.emplace_back(normal[index - 1]);
146 }
147
148 /* increase the length of current_material */
149 current_group->length += 3;
150 total_length += 3;
151 }
152
153 /* create new mesh object and initialize everything */
154 else if (keyword == g) {
155 std::string name;
156 stream >> name;
157
158 /* if current mesh has no data, replace its name */
159 if (current_mesh->vertex.empty()) {
160 current_mesh->name = name;
161 continue;
162 }
163
164 /* initialize total length */
165 total_length = 0;
166
167 /* create new mesh object */
168 meshes.emplace_back(Mesh(name));
169 current_mesh = &meshes.back();
170
171 /* create new mesh group */
172 current_mesh->groups.emplace_back<MeshGroup>({name, total_length, 0});
173 current_group = &current_mesh->groups.back();
174 }
175
176 /* use material */
177 else if (keyword == "usemtl") {
178 std::string name;
179 stream >> name;
180
181 /* if current mesh group has no data, replace its name */
182 if (current_group->length == 0) {
183 current_group->name = name;
184 continue;
185 }
186
187 /* create new mesh group */
188 current_mesh->groups.emplace_back<MeshGroup>({name, total_length, 0});
189 current_group = &current_mesh->groups.back();
190 }
191
192 /* ignore */
193 else {
194 stream.ignore(stream_max, '\n');
195 }
196 }
197
198 /* close file stream */
199 stream.close();
200
201 /* return the number of meshes */
202 return meshes;
203}

205std::vector<Material> Loader::load_mtl(const std::string& p) {
206 /* prepare the file stream */
207 std::ifstream stream;
208 stream.open(p, std::ifstream::in);
209 if (!stream) {
210 Error::set("Loader: Error reading from mtl file");
211 return std::vector<Material>();
212 }
213 size_t stream_max = std::numeric_limits<std::streamsize>::max();
214
215 /* initialize material pointer */
216 std::vector<Material> materials;
217 Material* current_material = nullptr;
218
219 /* read data by line */
220 while (!stream.eof()) {
221 std::string keyword;
222 stream >> keyword;
223
224 /* create a new material */
225 if (keyword == "newmtl") {
226 std::string name;
227 stream >> name;
228 materials.emplace_back(Material(name));
229 current_material = &materials.back();
230 }
231
232 /* material is not declared */
233 else if (current_material == nullptr) {}
234
235 /* diffuse color */
236 else if (keyword == "Kd") {
237 Vec3 kd;
238 stream >> kd.x >> kd.y >> kd.z;
239 current_material->color = kd;
240 }
241
242 /* emissive color */
243 else if (keyword == "Ke") {
244 Vec3 ke;
245 stream >> ke.x >> ke.y >> ke.z;
246 current_material->emissive = ke;
247 }
248
249 /* dissolve factor */
250 else if (keyword == "d") {
251 float d;
252 stream >> d;
253 current_material->alpha = d;
254 }
255
256 /* transparency factor */
257 else if (keyword == "tr") {
258 float tr;
259 stream >> tr;
260 current_material->alpha = 1 - tr;
261 }
262
263 /* not used in PBR material */
264
265 /* ambient color */
266 /* else if (keyword == "Ka") {} */
267
268 /* specular color */
269 /* else if (keyword == "Ks") {} */
270
271 /* transmission color */
272 /* else if (keyword == "Tf") {} */
273
274 /* specular exponent */
275 /* else if (keyword == "Ns") {} */
276
277 /* optical density */
278 /* else if (keyword == "Ni") {} */
279
280 /* sharpness value */
281 /* else if (keyword == "sharpness") {} */
282
283 /* ignore */
284 else {
285 stream.ignore(stream_max, '\n');
286 }
287 }
288
289 /* close file stream */
290 stream.close();
291
292 /* return the number of materials */
293 return materials;
294}
