159void Renderer::load_mesh(const Mesh* m) {
160 if (mesh_cache.count(m) != 0) return;
161 size_t size = m->groups.size();
162 auto p = mesh_cache.insert({m, std::make_unique<Gpu::VertexObject[]>(size)});
163 auto* vertex_object = p.first->second.get();
164 for (int i = 0; i < size; ++i) {
165 vertex_object[i].load(*m, i);
166 }
167}

169void Renderer::unload_mesh(const Mesh* m) {
170 mesh_cache.erase(m);
171}

173void Renderer::load_image(const Image* i) {
174 if (image_cache.count(i) != 0) return;
175 auto p = image_cache.insert({i, std::make_unique<Gpu::Texture>()});
176 auto* texture = p.first->second.get();
177 texture->init_2d(*i, Gpu::Texture::default_format(*i));
178 std::invoke(texture_callback, *texture);
179}

185void Renderer::load_scene(const Scene& s) {
186 /* load the meshes linked with instance */
187 for (auto& instance : s.to_instances()) {
188 auto* mesh = instance->mesh;
189 if (mesh != nullptr) load_mesh(mesh);
190 }
191
192 /* load the images linked with instance */
193 for (auto& material : s.get_materials()) {
194 for (int i = 0; i < 16; ++i) {
195 auto* image = material->get_image(i);
196 if (image != nullptr) load_image(image);
197 }
198 if (material->normal_map != nullptr) {
199 load_image(material->normal_map);
200 }
201 if (material->displacement_map != nullptr) {
202 load_image(material->displacement_map);
203 }
204 if (material->color_map != nullptr) {
205 load_image(material->color_map);
206 }
207 if (material->alpha_map != nullptr) {
208 load_image(material->alpha_map);
209 }
210 if (material->emissive_map != nullptr) {
211 load_image(material->emissive_map);
212 }
213 if (material->ao_map != nullptr) {
214 load_image(material->ao_map);
215 }
216 if (material->roughness_map != nullptr) {
217 load_image(material->roughness_map);
218 }
219 if (material->metalness_map != nullptr) {
220 load_image(material->metalness_map);
221 }
222 if (material->specular_map != nullptr) {
223 load_image(material->specular_map);
224 }
225 }

357void Renderer::update_probe(const Scene& s, const Vec3& p, ReflectionProbe& r) const {
358 /* set the viewport with the resolution of probe */
359 Gpu::set_viewport(Gpu::Rect(r.resolution, r.resolution));
360
361 /* disable the scissor test */
362 Gpu::disable_scissor_test();
363
364 /* prepare texture for probe */
365 if (!probe_map) {
366 probe_map = std::make_unique<Gpu::Texture>();
367 }
368 probe_map->init_cube(r.resolution, r.resolution, TEXTURE_R8G8B8A8_UNORM);
369 probe_map->set_filters(TEXTURE_LINEAR, TEXTURE_LINEAR);
370
371 /* prepare depth texture for rendering */
372 if (!probe_buffer) {
373 probe_buffer = std::make_unique<Gpu::RenderBuffer>();
374 }
375 probe_buffer->init(r.resolution, r.resolution, TEXTURE_D24_UNORM);
376
377 /* prepare frame buffer for probe */
378 if (!probe_target) {
379 probe_target = std::make_unique<Gpu::FrameBuffer>();
380 probe_target->draw_attachments({0});
381 }
382 probe_target->set_depth_attachment(*probe_buffer);
383
384 /* initialize camera for probe */
385 PerspCamera camera = PerspCamera(PI_2, 1, 0.1, 100);
386
387 for (int i = 0; i < 6; ++i) {
388 /* update camera for each side of cube */
389 camera.lookat(p, CUBE_DIRECTIONS[i], CUBE_UPS[i]);
390
391 /* update and activate render target */
392 probe_target->set_attachment(*probe_map, 0, 0, i);
393 Gpu::FrameBuffer::activate(probe_target.get());
394
395 /* clear depth buffer before rendering */
396 Gpu::clear(false, true, false);
397
398 /* render skybox loaded by renderer */
399 render_skybox_to_buffer(camera, FORWARD_RENDERING);
400
401 /* render the opaque objects in scene */
402 render_to_buffer(s, camera, FORWARD_RENDERING, false);
403
404 /* render the transparent objects in scene */
405 Gpu::set_depth_writemask(false);
406 render_to_buffer(s, camera, FORWARD_RENDERING, true);
407 Gpu::set_depth_writemask(true);
408 }
409
410 /* set back to the default render target */
411 Gpu::FrameBuffer::activate(nullptr);
412
413 /* update probe with probe texture */
414 r.load_texture(*probe_map);
415}

421Defines Renderer::define_material(const Material& m) {
422 Defines defines; /* create new defines */
423
424 /* check whether to use vertex color */
425 defines.set_if("USE_VERTEX_COLOR", m.vertex_color);
426
427 /* check whether to use normal map */
428 defines.set_if("USE_NORMAL_MAP", m.normal_map != nullptr);
429
430 /* check whether to use normal map in tangent space */
431 defines.set_if("IN_TANGENT_SPACE", m.normal_map != nullptr && m.tangent_space);
432
433 /* check whether to use normal map in object space */
434 defines.set_if("IN_OBJECT_SPACE", m.normal_map != nullptr && !m.tangent_space);
435
436 /* check whether to use displacement map */
437 defines.set_if("USE_DISPLACEMENT_MAP", m.displacement_map != nullptr);
438
439 /* check whether to use color map */
440 defines.set_if("USE_COLOR_MAP", m.color_map != nullptr && !m.map_with_alpha);
441
442 /* check whether to use color map with alpha channel */
443 defines.set_if("USE_COLOR_ALPHA_MAP", m.color_map != nullptr && m.map_with_alpha);
444
445 /* check whether to use alpha map */
446 defines.set_if("USE_ALPHA_MAP", m.alpha_map != nullptr);
447
448 /* check whether to use emissive map */
449 defines.set_if("USE_EMISSIVE_MAP", m.emissive_map != nullptr);
450
451 /* check whether to use ambient occlusion map */
452 defines.set_if("USE_AO_MAP", m.ao_map != nullptr);
453
454 /* check whether to use metalness map */
455 defines.set_if("USE_METALNESS_MAP", m.metalness_map != nullptr);
456
457 /* check whether to use roughness map */
458 defines.set_if("USE_ROUGHNESS_MAP", m.roughness_map != nullptr);
459
460 /* check whether to use specular map */
461 defines.set_if("USE_SPECULAR_MAP", m.specular_map != nullptr);
462
463 /* check whether to use light probe */
464 defines.set_if("USE_LIGHT_PROBE", m.light_probe != nullptr);
465
466 /* check whether to use reflection probe */
467 defines.set_if("USE_REFLECTION_PROBE", m.reflection_probe != nullptr);
468
469 return defines; /* return the defines */
470}

529void Renderer::set_light_uniforms(const Scene& s, const Gpu::Shader& shader) {
530 /* determines whether to enable shadow */
531 bool enable_shadow = false;
532
533 /* apply point lights in lighting */
534 size_t point_light_count = s.get_point_light_count();
535 for (int i = 0; i < point_light_count; ++i) {
536
537 /* pass the light information to shader */
538 auto& light = *s.get_point_light(i);
539 auto lights_i = Format::format("point_lights[{}]", i);
540 Vec3 light_color = light.color * light.intensity * PI;
541 shader.set_uniform_i(lights_i + ".visible", light.visible);
542 shader.set_uniform_v3(lights_i + ".position", light.position);
543 shader.set_uniform_v3(lights_i + ".color", light_color);
544 shader.set_uniform_f(lights_i + ".distance", light.distance);
545 shader.set_uniform_f(lights_i + ".decay", light.decay);
546 }
547
548 /* apply spot lights in lighting */
549 size_t spot_light_count = s.get_spot_light_count();
550 for (int i = 0; i < spot_light_count; ++i) {
551
552 /* pass the light information to shader */
553 auto& light = *s.get_spot_light(i);
554 std::string lights_i = Format::format("spot_lights[{}]", i);
555 Vec3 light_direction = -light.direction.normalize();
556 Vec3 light_color = light.color * light.intensity * PI;
557 float light_angle = cosf(light.angle);
558 float light_penumbra = cosf(light.angle * (1 - light.penumbra));
559 shader.set_uniform_i(lights_i + ".visible", light.visible);
560 shader.set_uniform_v3(lights_i + ".position", light.position);
561 shader.set_uniform_v3(lights_i + ".direction", light_direction);
562 shader.set_uniform_v3(lights_i + ".color", light_color);
563 shader.set_uniform_f(lights_i + ".distance", light.distance);
564 shader.set_uniform_f(lights_i + ".decay", light.decay);
565 shader.set_uniform_f(lights_i + ".angle", light_angle);
566 shader.set_uniform_f(lights_i + ".penumbra", light_penumbra);
567 shader.set_uniform_i(lights_i + ".cast_shadow", light.cast_shadow);
568
569 /* check whether the light cast shadow */
570 if (!light.cast_shadow) continue;
571 enable_shadow = true;
572
573 /* pass the shadow information to shader */
574 auto& shadow = light.shadow;
575 std::string shadows_i = Format::format("spot_lights[{}].shadow", i);
576 Mat4 view_proj = shadow.camera.projection * shadow.camera.viewing;
577 shader.set_uniform_i(shadows_i + ".type", shadow.type);
578 shader.set_uniform_i(shadows_i + ".map_id", shadow.get_unique_id());
579 shader.set_uniform_f(shadows_i + ".bias", shadow.bias);
580 shader.set_uniform_f(shadows_i + ".normal_bias", shadow.normal_bias);
581 shader.set_uniform_f(shadows_i + ".radius", shadow.radius);
582 shader.set_uniform_m4(shadows_i + ".proj", shadow.camera.projection);
583 shader.set_uniform_m4(shadows_i + ".view_proj", view_proj);
584 }
585
586 /* apply directional lights in lighting */
587 size_t directional_light_count = s.get_directional_light_count();
588 for (int i = 0; i < directional_light_count; ++i) {
589
590 /* pass the light information to shader */
591 auto& light = *s.get_directional_light(i);
592 std::string lights_i = Format::format("directional_lights[{}]", i);
593 Vec3 light_direction = -light.direction.normalize();
594 Vec3 light_color = light.color * light.intensity * PI;
595 shader.set_uniform_i(lights_i + ".visible", light.visible);
596 shader.set_uniform_v3(lights_i + ".direction", light_direction);
597 shader.set_uniform_v3(lights_i + ".color", light_color);
598 shader.set_uniform_i(lights_i + ".cast_shadow", light.cast_shadow);
599
600 /* check whether the light cast shadow */
601 if (!light.cast_shadow) continue;
602 enable_shadow = true;
603
604 /* pass the shadow information to shader */
605 auto& shadow = light.shadow;
606 std::string shadows_i = Format::format("directional_lights[{}].shadow", i);
607 Mat4 view_proj = shadow.camera.projection * shadow.camera.viewing;
608 shader.set_uniform_i(shadows_i + ".type", shadow.type);
609 shader.set_uniform_i(shadows_i + ".map_id", shadow.get_unique_id());
610 shader.set_uniform_f(shadows_i + ".bias", shadow.bias);
611 shader.set_uniform_f(shadows_i + ".normal_bias", shadow.normal_bias);
612 shader.set_uniform_f(shadows_i + ".radius", shadow.radius);
613 shader.set_uniform_m4(shadows_i + ".proj", shadow.camera.projection);
614 shader.set_uniform_m4(shadows_i + ".view_proj", view_proj);
615 }
616
617 /* apply hemisphere lights in lighting */
618 size_t hemisphere_light_count = s.get_hemisphere_light_count();
619 for (int i = 0; i < hemisphere_light_count; ++i) {
620
621 /* pass the light information to shader */
622 auto& light = *s.get_hemisphere_light(i);
623 std::string lights_i = Format::format("hemisphere_lights[{}]", i);
624 Vec3 light_sky_color = light.color * light.intensity * PI;
625 Vec3 light_ground_color = light.ground_color * light.intensity * PI;
626 shader.set_uniform_i(lights_i + ".visible", light.visible);
627 shader.set_uniform_v3(lights_i + ".direction", light.direction);
628 shader.set_uniform_v3(lights_i + ".sky_color", light_sky_color);
629 shader.set_uniform_v3(lights_i + ".ground_color", light_ground_color);
630 }
631
632 /* pass the shadow parameters to shader */
633 if (enable_shadow) Shadow::activate_texture(26);
634 shader.set_uniform_i("global_shadow.map", 26);
635 shader.set_uniform_v2("global_shadow.size", Shadow::get_resolution());
636
637 /* pass the linear fog parameters to shader */
638 auto* linear_fog = s.get_linear_fog();
639 if (linear_fog != nullptr) {
640 shader.set_uniform_i("linear_fog.visible", linear_fog->visible);
641 shader.set_uniform_v3("linear_fog.color", linear_fog->color);
642 shader.set_uniform_f("linear_fog.near", linear_fog->near);
643 shader.set_uniform_f("linear_fog.far", linear_fog->far);
644 }
645
646 /* pass the exp square fog parameters to shader */
647 auto* exp2_fog = s.get_exp2_fog();
648 if (exp2_fog != nullptr) {
649 shader.set_uniform_i("exp2_fog.visible", exp2_fog->visible);
650 shader.set_uniform_v3("exp2_fog.color", exp2_fog->color);
651 shader.set_uniform_f("exp2_fog.near", exp2_fog->near);
652 shader.set_uniform_f("exp2_fog.density", exp2_fog->density);
653 }
654}

656void Renderer::render_skybox_to_buffer(const Camera& c, int r) const {
657 /* initialize skybox */
658 [[maybe_unused]]
659 static bool inited = init_skybox();
660
661 /* set the depth & stencil test */
662 Gpu::disable_depth_test();
663 Gpu::disable_stencil_test();
664
665 /* disable blending & wireframe */
666 Gpu::disable_blending();
667 Gpu::disable_wireframe();
668
669 /* cull the front sides of faces */
670 Gpu::enable_culling();
671 Gpu::set_cull_side(FRONT_SIDE);
672
673 /* render cube texture seamlessly */
674 Gpu::enable_texture_cube_seamless();
675
676 /* fetch the skybox shader from shader lib */
677 Defines defines;
678 if (r == FORWARD_RENDERING) {
679 defines.set(define_tone_map(tone_map_mode));
680 } else {
681 defines.set("TONE_MAP", "linear_tone_map");
682 }
683 int type = skybox_map->get_type();
684 defines.set_if("USE_EQUIRECT", type == TEXTURE_2D);
685 defines.set_if("USE_CUBEMAP", type == TEXTURE_CUBE);
686 auto* shader = ShaderLib::fetch("Skybox", defines);
687
688 /* calculate transform matrice */
689 Mat4 viewing = c.viewing;
690 for (int i = 0; i < 3; ++i) {
691 viewing[i][3] = 0;
692 }
693 Mat4 view_proj = c.projection * viewing;
694
695 /* calculate exposure in tone mapping */
696 float exposure = 1;
697 if (r == FORWARD_RENDERING) exposure = tone_map_exposure;
698
699 /* render to the render target */
700 shader->use_program();
701 shader->set_uniform_m4("view_proj", view_proj);
702 shader->set_uniform_f("exposure", exposure);
703 shader->set_uniform_f("intensity", skybox_intensity);
704 shader->set_uniform_i("map", skybox_map->activate(0));
705 cube->attach(*shader);
706 cube->draw();

709void Renderer::render_to_buffer(const Scene& s, const Camera& c, int r, bool t) const {
710 /* create transform matrices & vectors */
711 Mat4 model;
712 Mat4 view = c.viewing;
713 Mat4 proj = c.projection;
714 Mat4 model_view;
715 Mat4 model_view_proj;
716 Mat3 normal_mat;
717 Vec3 camera_pos = c.position;
718
719 /* sort the vector of visible instances */
720 auto instances = s.to_visible_instances();
721 sort_instances(c, instances, t);
722
723 /* render all instances in sorted vector */
724 for (auto& instance : instances) {
725
726 /* get matrices from instance */
727 model = instance->matrix_global;
728 model_view = view * model;
729 model_view_proj = proj * model_view;
730 normal_mat = inverse_3x3(Mat3{
731 model[0][0], model[1][0], model[2][0],
732 model[0][1], model[1][1], model[2][1],
733 model[0][2], model[1][2], model[2][2],
734 });
735
736 /* get mesh from instance */
737 auto* mesh = instance->mesh;
738
739 /* check whether the scene is loaded */
740 if (mesh_cache.count(mesh) == 0) {
741 return Error::set("Renderer: Scene is not loaded");
742 }
743
744 /* get vertex objects from mesh cache */
745 auto* vertex_object = mesh_cache.at(mesh).get();
746 size_t group_size = mesh->groups.size();
747 for (int i = 0; i < group_size; ++i) {
748
749 /* get material from material groups */
750 auto& group = mesh->groups[i];
751 auto* material = s.get_material(group.name, instance);
752 if (material == nullptr) {
753 material = s.get_material(group.name, mesh);
754 }
755 if (material == nullptr) {
756 material = s.get_material(group.name);
757 }
758 if (material == nullptr) {
759 Error::set("Renderer: Material is not linked");
760 continue;
761 }
762
763 /* check whether the material is visible */
764 if (!material->visible) continue;
765
766 /* check whether the material is transparent or blending */
767 bool is_transparent = material->transparent || material->blending;
768 if (is_transparent != t) continue;
769
770 /* fetch the standard shader from shader lib */
771 auto* shader = static_cast<const Gpu::Shader*>(material->shader);
772 if (shader == nullptr) {
773 Defines defines = define_material(*material);
774 if (!t && r == DEFERRED_RENDERING) {
775 /* use defines in deferred rendering */
776 defines.set("DEFERRED_RENDERING");
777 } else {
778 /* use defines in forward rendering */
779 defines.set("FORWARD_RENDERING");
780 defines.set(define_scene(s));
781 defines.set(define_tone_map(tone_map_mode));
782 }
783 shader = ShaderLib::fetch("Standard", defines);
784 }
785
786 /* render vertex object with shader */
787 shader->use_program();
788 vertex_object[i].attach(*shader);
789
790 if (is_transparent || r == FORWARD_RENDERING) {
791 /* pass camera parameters to shader */
792 Mat4 inv_view_proj = inverse_4x4(c.projection * c.viewing);
793 shader->set_uniform_m4("inv_view_proj", inv_view_proj);
794
795 /* pass the tone mapping parameters to shader */
796 shader->set_uniform_f("exposure", tone_map_exposure);
797
798 /* pass the lights & fogs parameters to shader */
799 set_light_uniforms(s, *shader);
800 }
801
802 /* pass the renderer parameters to shader */
803 shader->set_uniform_m4("model" , model );
804 shader->set_uniform_m4("view" , view );
805 shader->set_uniform_m4("proj" , proj );
806 shader->set_uniform_m4("model_view" , model_view );
807 shader->set_uniform_m4("model_view_proj", model_view_proj);
808 shader->set_uniform_m3("normal_mat" , normal_mat );
809 shader->set_uniform_v3("camera_pos" , camera_pos );
810
811 /* pass the material parameters to shader */
812 shader->set_uniform_v3("color" , material->color );
813 shader->set_uniform_f("alpha_test" , material->alpha_test );
814 shader->set_uniform_f("alpha" , material->alpha );
815 shader->set_uniform_f("ao_intensity", material->ao_intensity);
816 shader->set_uniform_f("specular" , material->specular );
817 shader->set_uniform_f("metalness" , material->metalness );
818 shader->set_uniform_f("roughness" , material->roughness );
819
820 /* pass the emissive parameter to shader */
821 Vec3 emissive = material->emissive * material->emissive_intensity;
822 shader->set_uniform_v3("emissive", emissive);
823
824 /* pass the normal scale if use normal map */
825 if (material->normal_map != nullptr) {
826 shader->set_uniform_f("normal_scale", material->normal_scale);
827 }
828
829 /* pass the displacement scale if use displacement map */
830 if (material->displacement_map != nullptr) {
831 shader->set_uniform_f("displacement_scale", material->displacement_scale);
832 }
833
834 /* pass the images linked with material */
835 for (int j = 0; j < 16; ++j) {
836 auto* image = material->get_image(j);
837 if (image != nullptr) image_cache.at(image)->activate(j);
838 }
839 if (material->normal_map != nullptr) {
840 auto& map = image_cache.at(material->normal_map);
841 shader->set_uniform_i("normal_map", map->activate(16));
842 }
843 if (material->displacement_map != nullptr) {
844 auto& map = image_cache.at(material->displacement_map);
845 shader->set_uniform_i("displacement_map", map->activate(17));
846 }
847 if (material->color_map != nullptr) {
848 auto& map = image_cache.at(material->color_map);
849 shader->set_uniform_i("color_map", map->activate(18));
850 }
851 if (material->alpha_map != nullptr) {
852 auto& map = image_cache.at(material->alpha_map);
853 shader->set_uniform_i("alpha_map", map->activate(19));
854 }
855 if (material->emissive_map != nullptr) {
856 auto& map = image_cache.at(material->emissive_map);
857 shader->set_uniform_i("emissive_map", map->activate(20));
858 }
859 if (material->ao_map != nullptr) {
860 auto& map = image_cache.at(material->ao_map);
861 shader->set_uniform_i("ao_map", map->activate(21));
862 }
863 if (material->roughness_map != nullptr) {
864 auto& map = image_cache.at(material->roughness_map);
865 shader->set_uniform_i("roughness_map", map->activate(22));
866 }
867 if (material->metalness_map != nullptr) {
868 auto& map = image_cache.at(material->metalness_map);
869 shader->set_uniform_i("metalness_map", map->activate(23));
870 }
871 if (material->specular_map != nullptr) {
872 auto& map = image_cache.at(material->specular_map);
873 shader->set_uniform_i("specular_map", map->activate(24));
874 }
875
876 /* pass the reflection probe linked with material */
877 auto* ref_probe = static_cast<const ReflectionProbe*>(material->reflection_probe);
878 if (ref_probe != nullptr) {
879 int ref_lod = log2f(ref_probe->resolution);
880 shader->set_uniform_i("ref_map", ref_probe->activate(25));
881 shader->set_uniform_f("ref_lod", ref_lod);
882 shader->set_uniform_f("ref_intensity", ref_probe->intensity);
883 }
884
885 /* pass the custom uniforms linked with material */
886 if (material->uniforms != nullptr) {
887 shader->set_uniforms(*material->uniforms);
888 }
889
890 /* apply the depth test setting of material */
891 Gpu::apply_material_depth(*material);
892
893 /* apply the stencil test setting of material */
894 Gpu::apply_material_stencil(*material);
895
896 /* apply the wireframe setting of material */
897 Gpu::apply_material_wireframe(*material);
898
899 /* apply the blending setting of material */
900 Gpu::apply_material_blending(*material);
901
902 /* apply the render side setting of material */
903 Gpu::apply_material_side(*material);
904
905 /* set the blendings to render transparent objects correctly */
906 if (material->transparent) {
907 Gpu::enable_blending();
908 Gpu::set_blend_op(BLEND_ADD);
909 Gpu::set_blend_factor(FACTOR_SRC_ALPHA, FACTOR_ONE_MINUS_SRC_ALPHA);
910 }
911
912 /* cull back side of face if side is DOUBLE_SIDE */
913 if (is_transparent && material->side == DOUBLE_SIDE) {
914 Gpu::enable_culling();
915 Gpu::set_cull_side(BACK_SIDE);
916 }
917
918 /* draw the vertex object at group i */
919 vertex_object[i].draw();
920 }
921 }

1036void Renderer::sort_instances(const Camera& c, std::vector<const Instance*>& l, bool t) {
1037 using InstanceItem = std::pair<const Instance*, float>;
1038 std::vector<InstanceItem> sorted_instances;
1039 auto instances_size = l.size();
1040 sorted_instances.resize(instances_size);
1041 for (int i = 0; i < instances_size; ++i) {
1042 Vec3 position = l[i]->local_to_global({});
1043 sorted_instances[i].first = l[i];
1044 sorted_instances[i].second = (c.position - position).dot(c.direction);
1045 }
1046 auto compare = t ?
1047 [](const InstanceItem& a, const InstanceItem& b) -> bool {
1048 if (a.first->priority != b.first->priority) {
1049 return a.first->priority < b.first->priority;
1050 }
1051 return a.second > b.second;
1052 } :
1053 [](const InstanceItem& a, const InstanceItem& b) -> bool {
1054 if (a.first->priority != b.first->priority) {
1055 return a.first->priority < b.first->priority;
1056 }
1057 return a.second < b.second;
1058 };
1059 std::sort(sorted_instances.begin(), sorted_instances.end(), compare);
1060 for (int i = 0; i < instances_size; ++i) {
1061 l[i] = sorted_instances[i].first;
1062 }
1063}
